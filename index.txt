// ==================== ЦВЕТА АВАТАРОК ====================
const avatarColors = ['#e57373','#f06292','#ba68c8','#9575cd','#7986cb','#64b5f6','#4fc3f7','#4dd0e1','#4db6ac','#81c784','#ffb74d','#ff8a65'];
function getColorForName(name) {
    let hash = 0;
    for (let i = 0; i < name.length; i++) hash = name.charCodeAt(i) + ((hash << 5) - hash);
    return avatarColors[Math.abs(hash) % avatarColors.length];
}

// ==================== СОХРАНЕНИЕ / ЗАГРУЗКА ====================
function saveGame() {
    try {
        gameState.savedAt = Date.now();
        localStorage.setItem(SAVE_KEY, JSON.stringify(gameState));
        showSaveIndicator();
    } catch(e) { console.warn('Save failed', e); }
}

function loadSave() {
    try {
        const raw = localStorage.getItem(SAVE_KEY);
        return raw ? JSON.parse(raw) : null;
    } catch(e) { return null; }
}

function clearSave() {
    localStorage.removeItem(SAVE_KEY);
}

function showSaveIndicator() {
    els.saveIndicator.classList.add('show');
    setTimeout(() => els.saveIndicator.classList.remove('show'), 2000);
}

// Автосохранение каждые 30 сек
setInterval(() => { if (!gameState.gameOver && gameState.turnCount > 0) saveGame(); }, 30000);

els.manualSaveBtn.addEventListener('click', () => saveGame());

// ==================== ЗАГРУЗОЧНЫЙ ЭКРАН ====================
async function runLoadingScreen(hasSave) {
    const steps = hasSave
        ? ['Проверка сигнала...', 'Восстановление истории чата...', 'Расшифровка сообщений...', 'Соединение...']
        : ['Инициализация...', 'Поиск участников...', 'Подключение к серверу...', 'Готово'];
    
    for (let i = 0; i < steps.length; i++) {
        els.loadingText.textContent = steps[i];
        els.loadingBar.style.width = ((i + 1) / steps.length * 100) + '%';
        await delay(600);
    }
    await delay(400);
    els.loadingScreen.style.opacity = '0';
    els.loadingScreen.style.transition = 'opacity 0.5s';
    await delay(500);
    els.loadingScreen.style.display = 'none';
}

// ==================== СТАРТОВЫЙ ПОТОК ====================
async function init() {
    const save = loadSave();

    if (save && save.turnCount > 0) {
        // Есть сохранение
        await runLoadingScreen(true);
        const d = new Date(save.savedAt);
        els.saveInfoText.textContent = `Игрок: ${save.playerName} · Ход: ${save.turnCount}/10 · Сохранено: ${d.toLocaleTimeString('ru-RU', {hour:'2-digit',minute:'2-digit'})}`;
        els.continueModal.style.display = 'flex';
        
        els.continueBtn.onclick = async () => {
            els.continueModal.style.display = 'none';
            await restoreFromSave(save);
        };
        els.newGameBtn.onclick = () => {
            clearSave();
            els.continueModal.style.display = 'none';
            els.nameModal.style.display = 'flex';
        };
    } else {
        await runLoadingScreen(false);
        els.nameModal.style.display = 'flex';
    }
}

async function restoreFromSave(save) {
    Object.assign(gameState, save);

    // Рендерим историю сообщений
    // Сначала очищаем (оставим только системное приветствие)
    const msgs = els.chatMessages.querySelectorAll('.message-wrapper');
    msgs.forEach(m => { if (!m.classList.contains('system') || m === els.typingIndicator) {} });
    
    // Перерисовываем из истории
    for (const msg of gameState.history) {
        if (msg.type === 'system') {
            appendAuthorNarration(msg.text, msg.isHorror, true);
        } else {
            appendMessage(msg.sender, msg.text, msg.msgType, true);
        }
    }
    updateHorrorUI(gameState.horrorLevel);
    enableChat();
}

// ==================== СТАРТ НОВОЙ ИГРЫ ====================
els.startGameBtn.addEventListener('click', () => {
    const n = els.playerNameInput.value.trim();
    if (n) gameState.playerName = n;
    els.nameModal.style.display = 'none';

    if (!OPENROUTER_API_KEY || OPENROUTER_API_KEY.includes('ВСТАВЬТЕ')) {
        appendAuthorNarration('⚠️ API ключ не вставлен! Вставьте ключ OpenRouter в строку 7 кода.', false, true);
    }
    enableChat();
});

function enableChat() {
    els.chatInput.disabled = false;
    els.sendBtn.disabled = false;
    els.chatInput.focus();
}

// ==================== ОТПРАВКА СООБЩЕНИЯ ====================
async function sendPlayerMessage() {
    const text = els.chatInput.value.trim();
    if (!text || gameState.gameOver) return;

    appendMessage(gameState.playerName, text, 'player');
    els.chatInput.value = '';
    gameState.history.push({ sender: gameState.playerName, text, msgType: 'player', type: 'msg' });
    gameState.turnCount++;

    if (gameState.history.length > 24) gameState.history = gameState.history.slice(-24);
    saveGame();

    await runDirectorPipeline();
}

// ==================== ПРОМПТЫ ====================
function getPrompt1() {
    return `Ты Режиссёр 1 хоррор-игры. В чате 14 друзей застряли в заброшенном бункере.
Игрок: ${gameState.playerName}. Шаг: ${gameState.turnCount}/10. Уровень ужаса: ${gameState.horrorLevel}/100.
Учти уровень ужаса — чем выше, тем страшнее события.
Выдай JSON: { "author_narration": "описание нарастающей атмосферы", "plot_directive": "что должны сказать персонажи", "horror_delta": <число от -5 до +20> }`;
}

function getPrompt2() {
    return `Ты Режиссёр 2. Напиши диалоги по директиве.
Персонажи: Макс (лидер, паникует), Аня (медик, держится), Крис (скептик), Влад (юморит для вида), Марина (пропавшая).
Игрок: ${gameState.playerName}. Обращайся к нему/ней напрямую.
Выдай JSON: { "messages": [ {"sender": "Имя", "text": "Текст сообщения"} ] }`;
}

function getPrompt3() {
    return `Ты Режиссёр 3. Отполируй диалоги: живые эмоции, реализм, хоррор-атмосфера.
Шаг ${gameState.turnCount}/10, уровень ужаса ${gameState.horrorLevel}/100.
Если шаг >= 10 — завершить игру (game_over: true).
Выдай JSON: { "author_narration": "событие (если есть, иначе null)", "is_horror_event": true/false, "messages": [...], "game_over": false }`;
}

// ==================== КОНВЕЙЕР РЕЖИССЁРОВ ====================
async function runDirectorPipeline() {
    try {
        const histCtx = gameState.history
            .filter(m => m.type !== 'system')
            .slice(-16)
            .map(m => `${m.sender}: ${m.text}`).join('\n');

        showTyping();

        const call = (systemPrompt, userContent) =>
            fetch('https://openrouter.ai/api/v1/chat/completions', {
                method: 'POST',
                headers: { 'Authorization': `Bearer ${OPENROUTER_API_KEY}`, 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    model: 'google/gemini-2.5-flash',
                    messages: [{ role: 'system', content: systemPrompt }, { role: 'user', content: userContent }],
                    response_format: { type: 'json_object' }
                })
            }).then(r => r.json()).then(d => JSON.parse(d.choices[0].message.content));

        const p1 = await call(getPrompt1(), histCtx);
        const p2 = await call(getPrompt2(), `Событие: ${p1.author_narration}\nПлан: ${p1.plot_directive}\nЧат:\n${histCtx}`);
        const p3 = await call(getPrompt3(), `Черновик: ${JSON.stringify(p2)}`);

        hideTyping();

        // Обновляем уровень ужаса
        if (p1.horror_delta) {
            gameState.horrorLevel = Math.min(100, Math.max(0, gameState.horrorLevel + p1.horror_delta));
        }
        updateHorrorUI(gameState.horrorLevel);

        // Нарратив
        if (p3.author_narration) {
            appendAuthorNarration(p3.author_narration, p3.is_horror_event);
            gameState.history.push({ type: 'system', text: p3.author_narration, isHorror: p3.is_horror_event });
            if (p3.is_horror_event) triggerHorrorEffect();
            await delay(800);
        }

        // Сообщения НПС
        if (p3.messages) {
            for (const msg of p3.messages) {
                showTyping();
                await delay(Math.max(900, Math.min(msg.text.length * 25, 3000)));
                hideTyping();
                appendMessage(msg.sender, msg.text, 'npc');
                gameState.history.push({ sender: msg.sender, text: msg.text, msgType: 'npc', type: 'msg' });
            }
        }

        if (p3.game_over === true) {
            gameState.gameOver = true;
            appendAuthorNarration('📵 ИГРА ОКОНЧЕНА. Связь потеряна навсегда...', true);
            els.chatInput.disabled = true;
            els.sendBtn.disabled = true;
            triggerHorrorEffect(true);
            clearSave();
        } else {
            saveGame();
        }

    } catch (err) {
        hideTyping();
        console.error(err);
        appendAuthorNarration('⚡ Помехи в сети... сигнал прерван.', false);
    }
}

// ==================== РЕНДЕР СООБЩЕНИЙ ====================
function appendMessage(sender, text, type, silent = false) {
    const wrapper = document.createElement('div');
    wrapper.className = `message-wrapper ${type === 'npc' ? 'incoming' : 'outgoing'}`;

    const now = new Date();
    const timeStr = now.getHours().toString().padStart(2,'0') + ':' + now.getMinutes().toString().padStart(2,'0');

    if (type === 'npc') {
        const color = getColorForName(sender);
        const init = sender.charAt(0).toUpperCase();
        wrapper.innerHTML = `
            <div class="user-avatar" style="background-color:${color};">${init}</div>
            <div class="message incoming">
                <span class="sender-name" style="color:${color};">${sender}</span>
                <div class="message-text">${text}</div>
                <div class="time-wrapper">${timeStr}</div>
            </div>`;
    } else {
        wrapper.innerHTML = `
            <div class="message outgoing">
                <div class="message-text">${text}</div>
                <div class="time-wrapper">
                    ${timeStr}
                    <svg class="checks" viewBox="0 0 24 24"><path d="M18 7l-1.41-1.41-6.34 6.34 1.41 1.41L18 7zm4.24-1.41L11.66 16.17 7.48 12l-1.41 1.41L11.66 19l12-12-1.42-1.41zM.41 13.41L6 19l1.41-1.41L1.83 12 .41 13.41z"/></svg>
                </div>
            </div>`;
    }

    els.chatMessages.insertBefore(wrapper, els.typingIndicator);
    if (!silent) scrollToBottom();
    else bumpUnread();
}

function appendAuthorNarration(text, isHorror = false, silent = false) {
    const wrapper = document.createElement('div');
    wrapper.className = 'message-wrapper system';
    wrapper.innerHTML = `<div class="message system ${isHorror ? 'horror-event' : ''}">${text}</div>`;
    els.chatMessages.insertBefore(wrapper, els.typingIndicator);
    if (!silent) scrollToBottom();
    else bumpUnread();
}

// ==================== ВИЗУАЛЬНЫЕ ЭФФЕКТЫ УЖАСА ====================
function updateHorrorUI(level) {
    const pct = level + '%';
    els.horrorFill.style.width = pct;
    els.horrorValue.textContent = level + '%';
    els.horrorBar.style.width = level + '%';

    // Цвет шкалы
    if (level < 40) {
        els.horrorFill.style.background = 'linear-gradient(90deg, #f39c12, #e67e22)';
        els.horrorFill.style.boxShadow = 'none';
    } else if (level < 70) {
        els.horrorFill.style.background = 'linear-gradient(90deg, #e67e22, #e74c3c)';
        els.horrorFill.style.boxShadow = '0 0 6px rgba(231,76,60,0.4)';
    } else {
        els.horrorFill.style.background = 'linear-gradient(90deg, #c0392b, #ff1744)';
        els.horrorFill.style.boxShadow = '0 0 10px rgba(255,23,68,0.6)';
    }

    // Виньетка
    if (level >= 50) {
        els.redVignette.classList.add('active');
    } else {
        els.redVignette.classList.remove('active');
    }

    // Статус онлайн
    updateOnlineStatus(level);
}

function updateOnlineStatus(level) {
    const onlineCount = Math.max(1, Math.round(11 - (level / 100) * 8));
    const statusColor = level >= 70 ? 'danger-status' : '';
    els.chatStatus.className = 'chat-status ' + statusColor;
    els.chatStatus.textContent = level >= 80 
        ? `⚠️ ${onlineCount} в сети — сигнал слабый`
        : level >= 50 
            ? `${onlineCount} онлайн · есть помехи`
            : `${onlineCount} человек онлайн`;

    if (level >= 60) els.groupAvatar.classList.add('danger');
    else els.groupAvatar.classList.remove('danger');
}

function triggerHorrorEffect(final = false) {
    els.appContainer.classList.add('flash');
    setTimeout(() => els.appContainer.classList.remove('flash'), 1000);
    if (final) {
        els.redVignette.style.background = 'radial-gradient(ellipse at center, transparent 20%, rgba(192,57,43,0.6) 100%)';
    }
}

// ==================== УТИЛИТЫ ====================
function showTyping() {
    els.typingIndicator.style.display = 'flex';
    els.chatMessages.appendChild(els.typingIndicator);
    scrollToBottom();
}
function hideTyping() { els.typingIndicator.style.display = 'none'; }

let unreadCount = 0;
function bumpUnread() {
    const isAtBottom = els.chatMessages.scrollHeight - els.chatMessages.scrollTop - els.chatMessages.clientHeight < 100;
    if (!isAtBottom) {
        unreadCount++;
        els.unreadBadge.style.display = 'flex';
        els.unreadBadge.textContent = unreadCount;
        els.scrollBtn.classList.add('show');
    }
}

function scrollToBottom() {
    els.chatMessages.scrollTop = els.chatMessages.scrollHeight;
    unreadCount = 0;
    els.unreadBadge.style.display = 'none';
    els.scrollBtn.classList.remove('show');
}

els.chatMessages.addEventListener('scroll', () => {
    const isAtBottom = els.chatMessages.scrollHeight - els.chatMessages.scrollTop - els.chatMessages.clientHeight < 100;
    if (isAtBottom) {
        unreadCount = 0;
        els.unreadBadge.style.display = 'none';
        els.scrollBtn.classList.remove('show');
    } else {
        els.scrollBtn.classList.add('show');
    }
});
els.scrollBtn.addEventListener('click', scrollToBottom);

function delay(ms) { return new Promise(r => setTimeout(r, ms)); }
